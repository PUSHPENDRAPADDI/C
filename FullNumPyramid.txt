#include <stdio.h>
int main(){
    int num;
    printf("Enter number for pattern :-");
    scanf("%d", &num);
    for (int i = 0; i < num; i++)
    {
        for (int j = 0; j < (num-1) - i; j++)
        {
            printf("  ");
        }
        for (int j = 0; j < i; j++)
        {
            printf("  * ");
        }
        
        printf("\n");
    }
}
